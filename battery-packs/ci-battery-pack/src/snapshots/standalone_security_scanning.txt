── .github/workflows/audit.yml ──
name: Security audit

on:
  push:
    paths:
      - ".github/workflows/audit.yml"
      - ".cargo/audit.toml"
      - "**/Cargo.toml"
      - "**/Cargo.lock"
  schedule:
    - cron: "0 0 * * *"
  workflow_dispatch:

permissions:
  checks: write
  contents: read
  issues: write

jobs:
  security-audit:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@[..]
        with:
          persist-credentials: false
      - uses: rustsec/audit-check@[..]
        with:
          token: ${{ secrets.GITHUB_TOKEN }}
